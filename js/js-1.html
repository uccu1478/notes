Vue3 - VueSelect
html
:options放選項清單、label放文字(單一option是"{ title: "書名一" }"就放"title") 

<div id="app">
    <h1>Vue Select - Vue 3
    <br />{{param1}}
    <v-select :options="books" label="title" v-model="param1">
    <br />{{param2}}
    <v-select :options="someData" v-model="param2">
    <br />
</div>
js 

    var app;
    const books = [
        { title: "解決問題、創意湧現、高效表達的3×3系統思考法" },
        { title: "文明的邏輯：人類與風險的博弈" },
        { title: "逆思維：華頓商學院最具影響力的教授，突破人生盲點的全局思考" },
        { title: "垃圾之書：面對人類將被廢棄物所廢棄的事實與行動" },
    ];//這個是博客來某天的排行榜 不要在意書名
    const someData = { n1: '11', n2: '22', n3: '33' };
    
    $(function () {
        app = Vue.createApp({
            components: {
                vSelect: window["vue-select"] // Vue3在這用vue-select
            },
            data() {
                return {
                    books: books,
                    someData: Object.keys(someData),
                    param1: '',
                    param2: '',
                };
            },
        }).mount("#app");
    });
